15 struct CRhinoPickContext_MeshEdgeInfo
16 {
17  // edge list
18  // The i-th edge connects mesh vertices m_V[vi0] and m_V[vi1], where
19  // vi0 = m_edges[i*m_edge_stride] and vi1 = m_edges[i*m_edge_stride+1].
20  int m_edge_count;
21  int m_edge_stride;
22  const int* m_edges;
23 
24  // edge component index map
25  // If m_ci_map is null, then m_edges[] is in component index order.
26  // If m_ci_map is not null, then the meshtop_edge component index for
27  // the i-th edge is m_ci_map[i*m_ci_map_stride].
28  int m_edge_ci_map_stride;
29  const int* m_edge_ci_map;
30 
31  // vertext component index map
32  // m_vertex_ci_map[i*m_vertex_ci_map_stride] is the meshtop_vertex
33  // component index for the vertex ON_Mesh.m_V[i].
34  int m_vertex_ci_map_stride;
35  const int* m_vertex_ci_map;
36 };
