40  enum class Type : unsigned char
41  {
42  Unset = 0,
43 
44  /// <summary>
45  /// Reflection about a plane.
46  /// The symmetric object has 2 copies of the motif.
47  /// Points on the reflection plane are fixed.
48  /// </summary>
49  Reflect = 1,
50 
51  /// <summary>
52  /// Rotation around an axis.
53  /// The symmetric object has reflection copies of the motif.
54  /// Points on the axis are fixed.
55  /// </summary>
56  Rotate = 2,
57 
58  /// <summary>
59  /// Reflection and rotation alternate.
60  /// Rotation angle is 360/N degrees
61  /// The symmetric object has 2N copies of the motif.
62  /// Points on the rotation axis are fixed.
63  /// </summary>
64  ReflectAndRotate = 3,
65 
66  /// <summary>
67  /// General case inversion symmetry.
68  /// (inversion transformation)^2 = identity.
69  /// Det(inversion transformation) = -1.
70  /// The symmetric object has 2 copies of the motif.
71  /// </summary>
72  Inversion = 4,
73 
74  /// <summary>
75  /// General case cyclic symmtry (order >= 2)
76  /// (cyclic transformation)^N = identity.
77  /// When N is 2 or odd, Det(cyclic transformation) = 1.
78  /// When N is even and greater than 2, Det(cyclic transformation) = 1 or -1.
79  /// The symmetric object has N copies of the motif.
80  /// </summary>
81  Cyclic = 5,
82  };

141  /*
142  Description:
143  Rotation around an axis.
144  The symmetric object has reflection copies of the motif.
145  Points on the axis are fixed.
146  Parameters:
147  rotation_axis - [in]
148  rotation_count - [in]
149  rotation_count must be >= 2 and the rotation angle is (360/rotation_count) degrees.
150  symmetry_coordinates - [in]
151  object or world.
152  Example:
153  If the rotation axis is the z-axis and the order is N, then
154  then (x,y,z) -> (r*cos(a), r*sin(a), z) -> (r*cos(2*a), r*sin(2*a), z) -> ... -> (x,y,z),
155  where r = sqrt(x*x + y*y) and a = (360 degrees)/N.
156  Remarks:
157  CyclicTransform() = rotation.
158  InversionTransform() = identity.
159  */

178  object or world.
179  Example:
180  If the reflection_plane is the y-z plane, the rotation axis is the z-axis,
181  and rotation_count = 2,
182  then (x,y,z) -> (-x,y,z) -> (-x-y,z) -> (x,-y,z) -> (x,y,z).

184  InversionTransform() = reflection.
