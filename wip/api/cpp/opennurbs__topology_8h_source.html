35  /// <summary>
36  /// <para>ON_EdgeAttributeFlags are used to report attributes of single edge objects, like
37  /// ON_SubDEdge and ON_BrepEdge, and aggregate edge demographics in objects with topology
38  /// like ON_SubD, ON_Mesh and ON_Brep.</para>
39  /// <seealso cref="ON_BrepEdge::EdgeFlags"/>
40  /// <seealso cref="ON_SubDEdge::EdgeFlags"/>
41  /// <seealso cref="ON_Brep::AggregateEdgeFlags"/>
42  /// <seealso cref="ON_Mesh::AggregateEdgeFlags"/>
43  /// <seealso cref="ON_SubD::AggregateEdgeFlags"/>
44  /// </summary>
45  enum EdgeAttributes : unsigned int
46  {
47  ///<summary>
48  /// The Open bit is set when an edge has distinct start and end vertices.
49  /// Open and Closed are mutually exclusive edge attributes.
50  /// Note that an open edge can have zero length when two distict vertices
51  /// have identical locations and the edge's geometry is a zero length curve.
52  ///</summary>
53  Open = 1,
54 
55  ///<summary>
56  /// The Closed bit is set when an edge begins and ends at the same vertex.
57  /// Open and Closed are mutually exclusive edge attributes.
58  /// Note that a closed edge can have zero length or nonzero length.
59  ///</summary>
60  Closed = 2,
61 
62  ///<summary>
63  /// The ZeroLength edge has zero length. It can be open or closed.
64  /// ZeroLength and NonzeroLength are mutually exclusive edge attributes.
65  ///</summary>
66  ZeroLength = 4,
67 
68  ///<summary>
69  /// The NonzeroLength edge has non zero length. It can be open or closed.
70  /// ZeroLength and NonzeroLength are mutually exclusive edge attributes.
71  ///</summary>
72  NonzeroLength = 8,
73 
74  ///<summary>
75  /// The Wire edge has no faces.
76  /// Wire, Boundary, Interior, and Nonmanifold are mutually exclusive edge attributes.
77  ///</summary>
78  Wire = 16,
79 
80  ///<summary>
81  /// The Boundary edge has one face.
82  /// Wire, Boundary, Interior, and Nonmanifold are mutually exclusive edge attributes.
83  ///</summary>
84  Boundary = 32,
85 
86  ///<summary>
87  /// The Interior edge has two faces.
88  /// Wire, Boundary, Interior, and Nonmanifold are mutually exclusive edge attributes.
89  ///</summary>
90  Interior = 64,
91 
92  ///<summary>
93  /// The Nonmanifold edge has three or more faces.
94  /// Wire, Boundary, Interior, and Nonmanifold are mutually exclusive edge attributes.
95  ///</summary>
96  Nonmanifold = 128,
97 
98  ///<summary>
99  /// The InteriorOriented edge has two faces with compatible face orientations.
100  /// InteriorOriented and InteriorNotOriented are mutually exclusive interior edge attributes.
101  ///</summary>
102  InteriorOriented = 256,
103 
104  ///<summary>
105  /// The InteriorNotOriented edge has two faces with opposite face orientations.
106  /// InteriorOriented and InteriorNotOriented are mutually exclusive interior edge attributes.
107  ///</summary>
108  InteriorNotOriented = 512,
109 
110  ///<summary>
111  /// The InteriorSmooth edge has two faces with a guaranteed surface tangent continuity.
112  /// InteriorSmooth and InteriorCrease are mutually exclusive interior edge attributes.
113  ///</summary>
114  InteriorSmooth = 1024,
115 
116  ///<summary>
117  /// The InteriorCrease edge has two faces with a possible surface tangent discontinuity
118  /// InteriorSmooth and InteriorCrease are mutually exclusive interior edge attributes.
119  ///</summary>
120  InteriorCrease = 2048,
121 
122  ///<summary>
123  /// The InteriorTwoFaced edge has two distinct faces.
124  /// InteriorTwoFaced, InteriorSeam, and InteriorSlit are mutually exclusive interior edge attributes.
125  ///</summary>
126  InteriorTwoFaced = 4096,
127 
128  ///<summary>
129  /// The InteriorSeam edge has two faces that are identical and the edge is on the parametric boundary of the face's surface.
130  /// InteriorTwoFaced, InteriorSeam, and InteriorSlit are mutually exclusive interior edge attributes.
131  ///</summary>
132  ///<example>
133  /// Parametric surfaces that are cylinders are an example of a situation where seam edges occur.
134  ///</example>
135  InteriorSeam = 8192,
136 
137  ///<summary>
138  /// The InteriorSlit edge has two faces that are identical and the edges is not a seam.
139  /// InteriorTwoFaced, InteriorSeam, and InteriorSlit are mutually exclusive interior edge attributes.
140  ///</summary>
141  InteriorSlit = 16384,
142 
143 
144  ///<summary>
145  /// ON_ComponentAttributes::EdgeAttributes::Mask can be used to isolate EdgeAttributes bits
146  /// from an unsigned int bit field containing other information.
147  ///</summary>
148  ///<example>
149  /// Determine if two unsigned ints have identical ON_ComponentAttributes::EdgeAttributes settings.
150  ///<code>
151  /// unsigned int combined_flags1 = ...;
152  /// unsigned int combined_flags2 = ...;
153  /// unsigned int edge_flags1 = (ON_ComponentAttributes::EdgeAttributes::Mask &amp; combined_flags1);
154  /// unsigned int edge_flags2 = (ON_ComponentAttributes::EdgeAttributes::Mask &amp; combined_flags2);
155  /// if ( edge_flags1 == edge_flags1)
156  /// {
157  /// ... edges flags are identical ...
158  /// }
159  ///</code>
160  ///</example>
161  Mask = 0xFFFF
162  };
