134  ON_SimpleArray<TKey>& keys, TValue* values, TValue2* values2, int compareObjects(const TKey*, const TKey*), int compareObjectsContext(const TKey*, const TKey*, void*), void* context)
135  {

137  for (int i = 0; i < keys.Count(); i++) indices.Append(i);
138 
139  if (compareObjects != nullptr)
140  {
141  keys.Sort(ON::sort_algorithm::quick_sort, indices.Array(), compareObjects);
142  }
143  else
144  {
145  keys.Sort(ON::sort_algorithm::quick_sort, indices.Array(), compareObjectsContext, context);
146  }
147 
148  ON_SimpleArray<ON__UINT_PTR> back_indices(keys.Count()); back_indices.SetCount(keys.Count());
149  ON_SimpleArray<bool> done(keys.Count());
150 
151  for (ON__UINT_PTR i = 0; i < keys.Count(); i++)
152  {
153  back_indices[indices[i]] = i;
154  done.Append(false);
155  }
156 
157  unsigned long long i = 0;
158  while (i < keys.Count())
159  {
160  if (done[i])
161  {
162  i++;
163  }
164  else if (i != back_indices[i])
165  {
166  const ON__UINT_PTR& back_target = back_indices[i];
167  keys.Swap((int)i, (int)back_target);
168  std::swap(values[i], values[back_target]);
169  if (values2) std::swap(values2[i], values2[back_target]);
170 
171  done[back_indices[i]] = true;
172  back_indices.Swap((int)i, (int)back_indices[i]);
173  }
174  else i++;
175  }
176  }

272  template<class ForwardIt, class T>
273  ForwardIt lower_bound(ForwardIt first, ForwardIt last, const T& value, bool less_than(const T&, const T&, void*), void* context)
274  {
275  ForwardIt it;
276  ON__INT_PTR count;
277  count = last - first;
278 
279  while (count > 0)
280  {
281  it = first;
282  ON__INT_PTR step = count >> 1;
283  it += step;
284  if (less_than(*it, value, context))
285  {
286  first = ++it;
287  count -= step + 1;
288  }
289  else count = step;
290  }
291  return first;
292  }

316  template<class ForwardIt, class T>
317  inline std::pair<ForwardIt, ForwardIt> equal_range(
318  ForwardIt first, ForwardIt last, const T& value, bool less_than(const T&, const T&, void*), void* context)
319  {
320  ON__INT_PTR count = last - first;
321 
322  while (true)
323  {
324  if (count <= 0) { last = first; break; }
325 
326  ON__INT_PTR count2 = count >> 1;
327  ForwardIt middle = first + count2;
328  if (less_than(*middle, value, context))
329  {
330  first = (middle + 1);
331  count -= count2 + 1;
332  }
333  else if (less_than(value, *middle, context))
334  {
335  count = count2;
336  }
337  else
338  {
339  auto first2 = lower_bound(first, middle, value, less_than, context);
340  auto last2 = upper_bound(middle + 1, first + count, value, less_than, context);
341  last = last2;
342  first = first2;
343  break;
344  }
345  }
346 
347  return { first, last };
348  }
