356 private:
357  union {
358  // On little endian (Intel) computers, m_color has the same byte order
359  // as Windows COLORREF values.
360  // On little endian computers, m_color = 0xaabbggrr as an unsigned int value.
361  // On big endian computers, m_color = 0xrrggbbaa as an unsigned int value
362  // rr = red component 0-255
363  // gg = grean component 0-255
364  // bb = blue component 0-255
365  // aa = alpha 0-255. 0 means opaque, 255 means transparent.
366  unsigned int m_color = 0;
367 
368  // m_colorComponent is a 4 unsigned byte array in RGBA order
369  // red component = m_RGBA[ON_Color::RGBA_byte::kRed]
370  // grean component = m_RGBA[ON_Color::RGBA_byte::kGreen]
371  // blue component = m_RGBA[ON_Color::RGBA_byte::kBlue]
372  // alpha component = m_RGBA[ON_Color::RGBA_byte::kAlpha]
373  unsigned char m_RGBA[4];
374  };
